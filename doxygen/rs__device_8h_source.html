366 typedef enum rs2_calibration_status
367 {
368  // Anything >= 0 is not an issue
369  RS2_CALIBRATION_TRIGGERED = 0, // AC triggered and is active; conditions are valid
370  RS2_CALIBRATION_SPECIAL_FRAME = 1, // Special frame received; expect a frame-drop!
371  RS2_CALIBRATION_STARTED = 2, // Have all frames in hand; starting processing
372  RS2_CALIBRATION_NOT_NEEDED = 3, // Finished; existing calibration within tolerances; nothing done!
373  RS2_CALIBRATION_SUCCESSFUL = 4, // Finished; have new calibration in-hand
374 
375  RS2_CALIBRATION_RETRY = -1, // Initiating retry (asked for a new special frame)
376  RS2_CALIBRATION_FAILED = -2, // Unexpected: exception, device removed, stream stopped, etc.
377  RS2_CALIBRATION_SCENE_INVALID = -3, // Scene was not good enough for calibration; will retry
378  RS2_CALIBRATION_BAD_RESULT = -4, // Calibration finished, but results aren't good; will retry
379  RS2_CALIBRATION_BAD_CONDITIONS = -5, // Trigger was attempted but conditions (temp/APD) were invalid (still inactive)
380 
381  RS2_CALIBRATION_STATUS_FIRST = -5,
382  RS2_CALIBRATION_STATUS_LAST = 4,
383  RS2_CALIBRATION_STATUS_COUNT = RS2_CALIBRATION_STATUS_LAST - RS2_CALIBRATION_STATUS_FIRST + 1,
384 } rs2_calibration_status;
385 const char* rs2_calibration_status_to_string( rs2_calibration_status );
